    export function onRMDFieldsChange() {
        try {
            debugger;
            let ytdDistribution = FormScriptingHelper.getFieldValue(ct_accountFields.ct_rmdwithdrawals);
            let rmdRequirement = FormScriptingHelper.getFieldValue(ct_accountFields.ct_rmdrequirement);
            let rmdDateSatisfied = FormScriptingHelper.getDateValue(ct_accountFields.ct_rmddatesatisfied);

            // Only validate if RMD Date Satisfied is populated
            if (rmdDateSatisfied != null) {
                // Check if YTD Distribution is less than RMD Requirement
                if (ytdDistribution != null && rmdRequirement != null && ytdDistribution < rmdRequirement) {
                    // Clear the RMD Date Satisfied field
                    FormScriptingHelper.setDateField(ct_accountFields.ct_rmddatesatisfied, null);

                    // Show notification
                    FormScriptingHelper.setFormNotification("ERROR", "RMD was not satisfied and the field will be cleared. Double check that the requirement was met by reviewing RMD Requirement and YTD Distribution fields.", "rmdvalidation");
                } else {
                    // Clear the notification if validation passes
                    FormScriptingHelper.clearFormNotification("rmdvalidation");
                }
            }
        }
        catch (err) {
            if (typeof LogHelper.logUiException === "function") {
                LogHelper.logUiException(err, "onRMDFieldsChange");
            }
        }
    }

    function onct_rbChange()
    {

        let baTeam = FormScriptingHelper.CheckUserinTeamwithteamid("66c61e7c-4cff-e611-8104-fc15b4286c00");  // SECURITY TEAM Business Analyst (BA)
        let financeTeam = FormScriptingHelper.CheckUserinTeamwithteamid("f0a77772-4eff-e611-8104-fc15b4286c00");  // SECURITY TEAM Finance
        let wealthOpsTeam = FormScriptingHelper.CheckUserinTeamwithteamid("f33e7620-4dff-e611-8104-fc15b4286c00");  // SECURITY TEAM Wealth Operations

        var rb = FormScriptingHelper.getOptionSetValue(ct_accountFields.ct_rb);

        switch (rb) {
            case ct_accountOptions.ct_rb.R_Wealth_Redeemed:             //
                FormScriptingHelper.enableField(ct_accountFields.ct_raccountid);               
                FormScriptingHelper.makeFieldRequired(ct_accountFields.ct_raccountid);
                break;
            case ct_accountOptions.ct_rb.B_Wealth_MailedtoClient:       // 2
                FormScriptingHelper.enableField(ct_accountFields.ct_raccountid);
                FormScriptingHelper.makeFieldOptional(ct_accountFields.ct_raccountid);
                break;     
            default:
                FormScriptingHelper.disableField(ct_accountFields.ct_raccountid);
                FormScriptingHelper.makeFieldOptional(ct_accountFields.ct_raccountid);
                if (baTeam || financeTeam || wealthOpsTeam)
                {
                    FormScriptingHelper.clearLookupField(ct_accountFields.ct_raccountid);
                } 
                break;
        }
    }

    export function formatFidelityCharitable(f) {
        var r = /[^0-9]/g;
        var str = f.replace(r, '');
        str = str.substr(0, 7);
        return str;
    }
    export function formatFidelityDashes(f) {
        var r = /[\W_]+/g,
            npa = '',
            nxx = '';
        var str = f.replace(r, '');
        npa = str.substr(0, 3);
        nxx = str.substr(3, 6);
        str = npa + '-' + nxx;
        return str.toUpperCase();
    }
    export function formatPershing(f) {
        var r = /[\W_]+/g,
            npa = '';
        var str = f.replace(r, '');
        npa = str.substr(0, 9);
        str = npa;
        return str.toUpperCase();
    }
    export function formatSchwabDashes(f) {
        var r = /[^0-9]/g,
            npa = '',
            nxx = '';
        var str = f.replace(r, '');
        npa = str.substr(0, 4);
        nxx = str.substr(4, 4);
        return str = npa + '-' + nxx;
    }
    export function formatTDAmeritrade(f) {

    // either of the discreationary date fields onChange Event Handler
    export function onDiscretionaryDateChange() {
        try {
            let dtEligibility = FormScriptingHelper.getDateValue(ct_accountFields.ct_discretionarycompositeeligibilitydate);
            let dtDiscretionaryEnd = FormScriptingHelper.getDateValue(ct_accountFields.ct_discretionaryenddate);
            let dtInception = FormScriptingHelper.getDateValue(ct_accountFields.ct_legacydiscretionaryinception);
            if ((dtEligibility != null && dtDiscretionaryEnd != null) && dtEligibility >= dtDiscretionaryEnd) {
                FormScriptingHelper.openAlertDialog({
                    text: "Discretionary Composite Eligibility Date must be before Discretionary End Date."
                });
                FormScriptingHelper.setFocus(ct_accountFields.ct_discretionaryenddate);
            }
            if (dtEligibility == null && dtInception != null) {
                FormScriptingHelper.openAlertDialog({
                    text: "Update the Discretionary Composite Eligibility Date or clear Legacy Discretionary Inception Date."
                });
                FormScriptingHelper.setFocus(ct_accountFields.ct_legacydiscretionaryinception);
                return;
            }
            if (dtEligibility != null) {
                FormScriptingHelper.makeFieldRequired(ct_accountFields.ct_esg);
                FormScriptingHelper.makeFieldRequired(ct_accountFields.ct_clientrestrictions);
                FormScriptingHelper.enableField(ct_accountFields.ct_legacydiscretionaryinception);
            } else {
                FormScriptingHelper.disableField(ct_accountFields.ct_legacydiscretionaryinception);
                FormScriptingHelper.makeFieldOptional(ct_accountFields.ct_esg);
                FormScriptingHelper.makeFieldOptional(ct_accountFields.ct_clientrestrictions);
            }
        } catch (err) {
            if (typeof LogHelper.logUiException === "function") {
                LogHelper.logUiException(err, "onDiscretionaryDateChange");
            }
        }
    }

    export function HandleBPFStageChange(e) {
        try {
            let ct_accountid = FormScriptingHelper.getEntityId();
            let EventArgs = e.getEventArgs();
            let path = EventArgs.getDirection();
            let StageId = FormScriptingHelper.getactivestageid(); 
            let StageName = FormScriptingHelper.getactivestagename();

            if (previousStageName == "Custodian Account Setup" && StageName == "Billing") {
                var custodianValue = FormScriptingHelper.getFieldValue(ct_accountFields.ct_custodian);
                
                if (custodianValue == ct_accountOptions.ct_custodian.APEX) {
                    // start full billing automation for Zoe
                    var azureIntegrationsBaseURL = QueryHelper.getParameterValue("6f4fb73b-47f8-ef11-be20-000d3a8b2c12").ct_value;
                    var zoeStorageAccount = QueryHelper.getParameterValue("41c2753c-fdf9-ef11-be20-7c1e5258feba").ct_value;
                    var requestURL = azureIntegrationsBaseURL + "AddMessageToQueue";
                    var clientId = FormScriptingHelper.getLookupValue(ct_accountFields.ct_householdid);
                    var openDate = FormScriptingHelper.getDateValue(ct_accountFields.ct_dateopen);

                    var data = {
                        "StorageAccountName": zoeStorageAccount,
                        "QueueName": "zoemessages",
                        "MessageContent":{
                            "event_type": "CAPCONNECT_billing_automation",
                            "ct_accountid": ct_accountid,
                            "clientId": clientId.id,
                            "openDate": openDate
                        }
                    }

                    QueryHelper.webApiPost(requestURL, data)
                }
            }

        }
        catch (err) {
            if (typeof LogHelper.logUiException === "function") {
                LogHelper.logUiException(err, "HandleBPFStageChange");
            }
        }
    }

    //Show or Hide RAP info cards based on if lead consultant is in Minnetonka Office
    function HasAccountHoldings(accountid) {

        let uri = `
              <fetch version="1.0" output-format="xml-platform" mapping="logical" distinct="false" >
                <entity name="ct_holding" >
                <attribute name="ct_accountid" />
                <attribute name="ct_name" />
                <filter type="and" >
                    <condition attribute="ct_accountid" operator="eq" value="${accountid}" />
                </filter>
               <filter type="and" >
                    <condition attribute="statecode" operator="eq" value="${0}" />
                </filter>
                </entity>
            </fetch>
        `;

        let fetch = encodeURIComponent(uri);
        var url = QueryHelper.getWebApiUrl() + "ct_holdings?fetchXml=" + fetch;
        var result: Interface.IWebApiResult = QueryHelper.webApiRequest(url);

        if (result.value.length > 0) {
            return true;
        } else {
            return false;

                            let Accountname = FormScriptingHelper.removeBrackets(FormScriptingHelper.getEntityId());

                            if (Number != null && Number != undefined) {

                            }

                                if (accname != null && accname != undefined) {
                                    entity["ct_accountidname"] = accname;

                                entityName: ct_tradeflowrequestFields.entityname,
                                openInNewWindow: true
                            }, entity);

                        }
                    },

                );
            }


        }
        catch (err) {

            if (typeof LogHelper.logUiException === "function") {
                LogHelper.logUiException(err, "onCustodianChange");
            }
        }

        // Composite Tracking tab
        try {
            let dtEligibility = FormScriptingHelper.getDateValue(ct_accountFields.ct_discretionarycompositeeligibilitydate);
            let dtInception = FormScriptingHelper.getDateValue(ct_accountFields.ct_legacydiscretionaryinception);
            let dtDiscretionaryEnd = FormScriptingHelper.getDateValue(ct_accountFields.ct_discretionaryenddate);
            if ((dtEligibility != null && dtInception != null) && dtEligibility <= dtInception) {
                FormScriptingHelper.openAlertDialog({
                    text: "Legacy Discretionary Inception must be less than Discretionary Composite Eligibility Date."
                });
                FormScriptingHelper.setFocus(ct_accountFields.ct_legacydiscretionaryinception);
                executionObj.getEventArgs().preventDefault();
                return;
            }
            if ((dtEligibility == null && dtInception != null)) {
                FormScriptingHelper.openAlertDialog({
                    text: "Discretionary Composite Eligibility Date must be populated if Legacy Discretionary Inception exists."
                });
                FormScriptingHelper.setFocus(ct_accountFields.ct_discretionarycompositeeligibilitydate);
                executionObj.getEventArgs().preventDefault();
                return;
            }
            if ((dtEligibility != null && dtDiscretionaryEnd != null) && dtEligibility >= dtDiscretionaryEnd) {
                FormScriptingHelper.openAlertDialog({
                    text: "Discretionary End Date must be greater than Discretionary Composite Eligibility Date."
                });
                FormScriptingHelper.setFocus(ct_accountFields.ct_discretionaryenddate);
                executionObj.getEventArgs().preventDefault();
                return;
            }
        } catch (err) {
            if (typeof LogHelper.logUiException === "function") {
                LogHelper.logUiException(err, "onSave CompositeTracking");
            }

        }

        // Validate account number format
        try {
            // Get the custodian 
            var custodianValue = FormScriptingHelper.getFieldValue(ct_accountFields.ct_custodian);
            let accountId = FormScriptingHelper.getFieldValue(ct_accountFields.ct_accountnumber);
            let accountType = FormScriptingHelper.getOptionSetValue(ct_accountFields.ct_accounttype);
            //set an IF or switch on custodian.  If held away don't move on.  Else clean the field and add dashes or format as necessary.
            
            if (custodianValue != ct_accountOptions.ct_custodian.HeldAway && accountId != null) {
                let formattedAccountId = '';
                var valid = true;
                //Strip the field of any non-alphanumeric regardless of how it was input.  
                switch (custodianValue) {
                    case ct_accountOptions.ct_custodian.Fidelity:   //Fidelity - xxx-xxxxxx alphanumeric(10)
                        if (accountType = ct_accountOptions.ct_accounttype.Charitable_DonorAdvisedFund_) {
                            // For Charitable accounts, 7 digit account number without dashes
                            formattedAccountId = formatFidelityCharitable(accountId);
                            if (formattedAccountId.length < 7) {
                                valid = false;
                            }
                            break;
                        }
                        else {
                            formattedAccountId = formatFidelityDashes(accountId);
                            if (formattedAccountId.length < 10) {
                                valid = false;
                            }
                            break;
                        }
                    case ct_accountOptions.ct_custodian.Pershing:   //PAS/Pershing - xxxxxxxx - alphanumeric(9)
                        formattedAccountId = formatPershing(accountId);
                        if (formattedAccountId.length < 9) {
                            valid = false;
                        }
                        break;
                    case ct_accountOptions.ct_custodian.Schwab:     //Schwab - xxxx-xxxx numbers(9)
                        formattedAccountId = formatSchwabDashes(accountId);
                        if (formattedAccountId.length < 9) {
                            valid = false;
                        }
                        break;
                    case ct_accountOptions.ct_custodian.TDAmeritrade://TD Ameritrade - xxxxxxxxx numbers(9)
                        formattedAccountId = formatTDAmeritrade(accountId);
                        if (formattedAccountId.length < 9) {
                            valid = false;
                        }
                        break;
                    case ct_accountOptions.ct_custodian.APEX:       //APEX - xxxxxxxx alphanumeric(8)
                        // Check if accountId is alphanumeric, exactly 8 characters long, and contains no dashes
                        if (!/^[a-zA-Z0-9]{8}$/.test(accountId)) {
                            valid = false;
                        }
                        break;
                    default:
                        break;
                }
                //FormScriptingHelper.setTextboxValue("ct_accountnumber", formattedAccountId);
                if (!valid) {
                    FormScriptingHelper.openAlertDialog({ text: "Invalid account number format for Custodian" });
                    executionObj.getEventArgs().preventDefault();
                    return;
                }
            }

        } catch (err) {

            if (typeof LogHelper.logUiException === "function") {
                //display the message  
                FormScriptingHelper.setFieldNotification("ct_accountnumber", "Invalid account number format for Custodian");
                LogHelper.logUiException(err, "onAccountChange");
            }
        }
    }

    //ct_contract field behaviors
    export function contractRules() {
        try {
            var custodianValue = FormScriptingHelper.getFieldValue(ct_accountFields.ct_custodian);
            var currentContractValue = FormScriptingHelper.getFieldValue(ct_accountFields.ct_contract);
            FormScriptingHelper.makeFieldRequired(ct_accountFields.ct_contract);
            if (custodianValue == ct_accountOptions.ct_custodian.HeldAway) {
                FormScriptingHelper.addOptionSetOption(ct_accountFields.ct_contract, { text: ct_accountOptions.ct_contractMap[ct_accountOptions.ct_contract.Noneoftheabove], value: ct_accountOptions.ct_contract.Noneoftheabove });
                FormScriptingHelper.setOptionSetValue(ct_accountFields.ct_contract, ct_accountOptions.ct_contract.Noneoftheabove);
                FormScriptingHelper.makeFieldRequired(ct_accountFields.ct_contract);
                FormScriptingHelper.disableField(ct_accountFields.ct_contract);
            }
            else if (custodianValue == ct_accountOptions.ct_custodian.Pershing || custodianValue == ct_accountOptions.ct_custodian.Fidelity || custodianValue == ct_accountOptions.ct_custodian.Schwab || custodianValue == ct_accountOptions.ct_custodian.APEX) {
                FormScriptingHelper.removeOptionSetOptionByValue(ct_accountFields.ct_contract, ct_accountOptions.ct_contract.Noneoftheabove);
                if (currentContractValue != ct_accountOptions.ct_contract.Noneoftheabove) {
                    FormScriptingHelper.setOptionSetValue(ct_accountFields.ct_contract, currentContractValue);
                }
                else {
                    FormScriptingHelper.setOptionSetValue(ct_accountFields.ct_contract, null);
                }
                FormScriptingHelper.makeFieldRequired(ct_accountFields.ct_contract);
                FormScriptingHelper.enableField(ct_accountFields.ct_contract);
            }
            else {
                if (currentContractValue != null) {
                    FormScriptingHelper.addOptionSetOption(ct_accountFields.ct_contract, { text: ct_accountOptions.ct_contractMap[ct_accountOptions.ct_contract.Noneoftheabove], value: ct_accountOptions.ct_contract.Noneoftheabove });
                    FormScriptingHelper.setOptionSetValue(ct_accountFields.ct_contract, currentContractValue);
                    FormScriptingHelper.enableField(ct_accountFields.ct_contract);
                }
            }

            if ((FormScriptingHelper.getOptionSetValue(ct_accountFields.ct_rmdoverrideeligibility) == ct_accountOptions.ct_rmdoverrideeligibility.Yes) || (FormScriptingHelper.getFieldValue(ct_accountFields.ct_rmdeligiblity) == true)) {
                FormScriptingHelper.enableField("ct_rmddistributionmethod");
            } else {
                FormScriptingHelper.disableField("ct_rmddistributionmethod");
            }
        }
        catch (err) {
            if (typeof LogHelper.logUiException === "function") {
                LogHelper.logUiException(err, "toggleRMDOptionset");
            }
        }
    }

    export function onAccountRegistrationChange() {

        try {
            if (FormScriptingHelper.getOptionSetValue(ct_accountFields.ct_accounttype) == ct_accountOptions.ct_accounttype.InheritedIRA || FormScriptingHelper.getOptionSetValue(ct_accountFields.ct_accounttype) == ct_accountOptions.ct_accounttype.InheritedRothIRA) {
                FormScriptingHelper.setOptionSetValue(ct_accountFields.ct_rmdoverrideeligibility, ct_accountOptions.ct_rmdoverrideeligibility.Yes, true);
            }
        }
        catch (err) {
            if (typeof LogHelper.logUiException === "function") {
                LogHelper.logUiException(err, "onAccountRegistrationChange");
            }
        }

    }

    export function showSchwabAcctRestrictions() {

        var custodian = FormScriptingHelper.getOptionSetValue(ct_accountFields.ct_custodian);

        if (custodian == ct_accountOptions.ct_custodian.Schwab) {
            FormScriptingHelper.showSection("AccountInfo", "SchwabAccountRestrictions");
            var restrictions = FormScriptingHelper.getFieldValue("ct_schwabaccountrestrictions");
            if (restrictions != null && restrictions.length > 0) {
                FormScriptingHelper.setFormNotification("ERROR", "Schwab - Account Has Restriction(s)", "restrictionsnotification");
            }
        } else {
            FormScriptingHelper.hideSection("AccountInfo", "SchwabAccountRestrictions");
        }
    }

    export function showTradingRequestsTab() {
        try {
            const TradingRequestTab = "ct_tradeactionrequests_tab";
            let hasRole = false; 
            Xrm.Utility.getGlobalContext().userSettings.roles.forEach(
                role => { if(role.name == "Trading Canvas App Access") { hasRole = true;} }
            );
            if (hasRole)
            {
                SetIFrameSorce();
                FormScriptingHelper.showTab(TradingRequestTab);
            }
            else
            {
                FormScriptingHelper.hideTab(TradingRequestTab);
            }
        }
        catch (err) {
            if (typeof LogHelper.logUiException === "function") {
                LogHelper.logUiException(err, "showTradingRequestsTab()");
            }
        }
    }

    export function SetIFrameSorce()
    {
        try {
            const url = QueryHelper.getSystemParameter("TradingAppURL").ct_value;

            FormScriptingHelper.getFormContext()
                .getControl<Xrm.Controls.IframeControl>("IFRAME_ct_tradeoperationsrequest")
                .setSrc(url);
        }
        catch (err) {
            if (typeof LogHelper.logUiException === "function") {
                LogHelper.logUiException(err, "SetIFrameSorce()");
            }
        }
    }
